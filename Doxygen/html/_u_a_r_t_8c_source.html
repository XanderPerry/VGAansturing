28void usart2_init(void)
29{
30 // --- Clock Configuration ---
31
32 // Enable GPIOA clock (AHB1)
33 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN; // Bit 0 is GPIOAEN
34
35 // Enable USART2 clock (APB1)
36 RCC->APB1ENR |= RCC_APB1ENR_USART2EN; // Bit 17 is USART2EN
37
38
39 // --- GPIO Configuration (PA3=TX, PA2=RX) ---
40
41 // Configure PA2 and PA3 for Alternate Function (AF) mode (Mode = 10)
42 // Clear bits 4-5 (PA2) and 6-7 (PA3) and then set them to 10b
43 GPIOA->MODER &= ~( (3 << 4) | (3 << 6) );
44 GPIOA->MODER |= ( (2 << 4) | (2 << 6) );
45
46 // Configure Alternate Function Register Low (AFRL) to AF7 for PA2 and PA3
47 // AF7 is 0111
48 GPIOA->AFR[0] &= ~( (0xF << 8) | (0xF << 12) ); // Clear bits 8-15
49 GPIOA->AFR[0] |= ( (7 << 8) | (7 << 12) ); // Set AF7 (for PA2 and PA3)
50
51
52 // --- GPIO Pull-up/Pull-down Configuration ---
53
54 // Set PA2 (RX pin) to Pull-up (Mode = 01). Bits 4-5 of PUPDR. <-- CHANGE: PA2 is now RX
55 // Clear bits 4-5 (PA2)
56 GPIOA->PUPDR &= ~(3 << 4);
57 // Set bits 4-5 to 01b (Pull-up)
58 GPIOA->PUPDR |= (1 << 4);
59
60 // --- GPIO Output Speed Configuration ---
61
62 // Configure PA2 (RX) and PA3 (TX) for High Speed (Mode = 10b)
63 // Clear bits 4-5 (PA2) and 6-7 (PA3)
64 GPIOA->OSPEEDR &= ~( (3 << 4) | (3 << 6) );
65 // Set bits 4-5 and 6-7 to 10b (High Speed)
66 GPIOA->OSPEEDR |= ( (2 << 4) | (2 << 6) );
67
68
69 // --- USART Configuration ---
70
71 // Set Baud Rate to 115200 (using the correct value for PCLK1 = 42MHz)
72 // BRR = 0x16E (22.89) <-- FIX FOR GIBBERISH ISSUE
73 USART2->BRR = 0x111;
74
75 // Enable Transmitter (TE, Bit 3) and Receiver (RE, Bit 2)
76 USART2->CR1 |= (USART_CR1_TE | USART_CR1_RE);
77
78 // Enable USART (UE, Bit 13)
79 USART2->CR1 |= USART_CR1_UE;
80}

152void USART2_IRQHandler(void)
153{
154 // Check if the cause of the interrupt is the Read Data Register Not Empty (RXNE) flag
155 if (USART2->SR & USART_SR_RXNE)
156 {
157 uint8_t data = (uint8_t)(USART2->DR & 0xFF); // Read data and clear RXNE flag
158
159 // Calculate the next head index (write index)
160 uint16_t next_head = (uart_rx_head + 1) % UART_RX_BUFFER_SIZE;
161
162 // Check for buffer overrun (if the next head equals the tail)
163 if (next_head != uart_rx_tail)
164 {
165 // Store the data in the buffer
166 uart_rx_buffer[uart_rx_head] = data;
167
168 // Update the head index
169 uart_rx_head = next_head;
170
171 // Check if the received character is a Line Feed (LF)
172 if (data == LF_CHAR)
173 {
174 // Set the flag to notify the main loop that a line is ready
175 uart_rx_line_ready = true;
176 }
177 }
178 // If overrun, the character is discarded, and the head/tail are not updated.
179 }
180}

188bool usart2_read_line(char *dest, uint16_t max_len)
189{
190 // Check if the ready flag is set
191 if (!uart_rx_line_ready)
192 {
193 return false;
194 }
195
196 uint16_t i = 0;
197 bool lf_found = false;
198
199 // Read while buffer is not empty AND destination buffer has space
200 while (uart_rx_tail != uart_rx_head && i < (max_len - 1))
201 {
202 uint8_t data = uart_rx_buffer[uart_rx_tail];
203
204 // Update the tail index (read index)
205 uart_rx_tail = (uart_rx_tail + 1) % UART_RX_BUFFER_SIZE;
206
207 // Store data in the destination buffer
208 dest[i++] = data;
209
210 // Check for the Line Feed character
211 if (data == LF_CHAR)
212 {
213 lf_found = true;
214 break; // Stop reading after the LF is found
215 }
216 }
217
218 // Null-terminate the destination string
219 dest[i] = '\0';
220
221 // If the LF was found, clear the flag and return success
222 if (lf_found)
223 {
224 uart_rx_line_ready = false;
225 return true;
226 }
227
228 return false;
229}
