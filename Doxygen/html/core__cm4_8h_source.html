1025static __INLINE void NVIC_SetPriorityGrouping(uint32_t PriorityGroup)
1026{
1027 uint32_t reg_value;
1028 uint32_t PriorityGroupTmp = (PriorityGroup & (uint32_t)0x07); /* only values 0..7 are used */
1029
1030 reg_value = SCB->AIRCR; /* read old register configuration */
1031 reg_value &= ~(SCB_AIRCR_VECTKEY_Msk | SCB_AIRCR_PRIGROUP_Msk); /* clear bits to change */
1032 reg_value = (reg_value |
1033 ((uint32_t)0x5FA << SCB_AIRCR_VECTKEY_Pos) |
1034 (PriorityGroupTmp << 8)); /* Insert write key and priorty group */
1035 SCB->AIRCR = reg_value;
1036}

1189static __INLINE uint32_t NVIC_EncodePriority (uint32_t PriorityGroup, uint32_t PreemptPriority, uint32_t SubPriority)
1190{
1191 uint32_t PriorityGroupTmp = (PriorityGroup & 0x07); /* only values 0..7 are used */
1192 uint32_t PreemptPriorityBits;
1193 uint32_t SubPriorityBits;
1194
1195 PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1196 SubPriorityBits = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1197
1198 return (
1199 ((PreemptPriority & ((1 << (PreemptPriorityBits)) - 1)) << SubPriorityBits) |
1200 ((SubPriority & ((1 << (SubPriorityBits )) - 1)))
1201 );
1202}
