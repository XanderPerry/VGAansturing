34//--------------------------------------------------------------
35// Init VGA-Module
36//--------------------------------------------------------------
37void UB_VGA_Screen_Init(void)
38{
39 uint16_t xp,yp;
40
41 VGA.hsync_cnt=0;
42 VGA.start_adr=0;
43 VGA.dma2_cr_reg=0;
44
45 // RAM init total black
46 for(yp=0;yp<VGA_DISPLAY_Y;yp++) {
47 for(xp=0;xp<(VGA_DISPLAY_X+1);xp++) {
48 VGA_RAM1[(yp*(VGA_DISPLAY_X+1))+xp]=0;
49 }
50 }
51
52 // init IO-Pins
53 P_VGA_InitIO();
54 // init Timer
55 P_VGA_InitTIM();
56 // init DMA
57 P_VGA_InitDMA();
58 // init Interrupts
59 P_VGA_InitINT();
60
61 //-----------------------
62 // Register swap and safe
63 //-----------------------
64 // content of CR-Register read and save
65 VGA.dma2_cr_reg=DMA2_Stream5->CR;
66}

69//--------------------------------------------------------------
70// fill the DMA RAM buffer with one color
71//--------------------------------------------------------------
72void UB_VGA_FillScreen(uint8_t color)
73{
74 uint16_t xp,yp;
75
76 for(yp=0;yp<VGA_DISPLAY_Y;yp++) {
77 for(xp=0;xp<VGA_DISPLAY_X;xp++) {
78 UB_VGA_SetPixel(xp,yp,color);
79 }
80 }
81}

84//--------------------------------------------------------------
85// put one Pixel on the screen with one color
86// Important : the last Pixel+1 from every line must be black (don't know why??)
87//--------------------------------------------------------------
88void UB_VGA_SetPixel(uint16_t xp, uint16_t yp, uint8_t color)
89{
90 if (color == 0x01) return; // skip background pixel
91 if(xp>=VGA_DISPLAY_X) xp=0;
92 if(yp>=VGA_DISPLAY_Y) yp=0;
93
94 // Write pixel to ram
95 VGA_RAM1[(yp*(VGA_DISPLAY_X+1))+xp]=color;
96}

172void P_VGA_InitTIM(void)
173{
174 TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure;
175 TIM_OCInitTypeDef TIM_OCInitStructure;
176
177
178 //---------------------------------------------
179 // init of Timer1 for
180 // Pixeldata via DMA
181 //---------------------------------------------
182
183 // Clock enable
184 RCC_APB2PeriphClockCmd(RCC_APB2Periph_TIM1, ENABLE);
185
186 // Timer1 init
187 TIM_TimeBaseStructure.TIM_Period = VGA_TIM1_PERIODE;
188 TIM_TimeBaseStructure.TIM_Prescaler = VGA_TIM1_PRESCALE;
189 TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
190 TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
191 TIM_TimeBaseInit(TIM1, &TIM_TimeBaseStructure);
192
193
194 //---------------------------------------------
195 // init Timer2
196 // CH4 for HSYNC-Signal
197 // CH3 for DMA Trigger start
198 //---------------------------------------------
199
200 // Clock enable
201 RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM2, ENABLE);
202
203 // Timer2 init
204 TIM_TimeBaseStructure.TIM_Period = VGA_TIM2_HSYNC_PERIODE;
205 TIM_TimeBaseStructure.TIM_Prescaler = VGA_TIM2_HSYNC_PRESCALE;
206 TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
207 TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
208 TIM_TimeBaseInit(TIM2, &TIM_TimeBaseStructure);
209
210 // Timer2 Channel 3 ( for DMA Trigger start)
211 TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
212 TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
213 TIM_OCInitStructure.TIM_Pulse = VGA_TIM2_HTRIGGER_START-VGA_TIM2_DMA_DELAY;
214 TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_Low;
215 TIM_OC3Init(TIM2, &TIM_OCInitStructure);
216 TIM_OC3PreloadConfig(TIM2, TIM_OCPreload_Enable);
217
218 // Timer2 Channel 4 (for HSYNC)
219 TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
220 TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
221 TIM_OCInitStructure.TIM_Pulse = VGA_TIM2_HSYNC_IMP;
222 TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_Low;
223 TIM_OC4Init(TIM2, &TIM_OCInitStructure);
224 TIM_OC4PreloadConfig(TIM2, TIM_OCPreload_Enable);
225
226
227 //---------------------------------------------
228 // enable all Timers
229 //---------------------------------------------
230
231 // Timer1 enable
232 TIM_ARRPreloadConfig(TIM1, ENABLE);
233
234 // Timer2 enable
235 TIM_ARRPreloadConfig(TIM2, ENABLE);
236 TIM_Cmd(TIM2, ENABLE);
237
238}

244void P_VGA_InitINT(void)
245{
246 NVIC_InitTypeDef NVIC_InitStructure;
247
248 //---------------------------------------------
249 // init from DMA Interrupt
250 // for TransferComplete Interrupt
251 // DMA2, Stream5, Channel6
252 //---------------------------------------------
253
254 DMA_ITConfig(DMA2_Stream5, DMA_IT_TC, ENABLE);
255
256 // NVIC config
257 NVIC_InitStructure.NVIC_IRQChannel = DMA2_Stream5_IRQn;
258 NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0;
259 NVIC_InitStructure.NVIC_IRQChannelSubPriority = 0;
260 NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
261 NVIC_Init(&NVIC_InitStructure);
262
263
264 //---------------------------------------------
265 // init of Timer2 Interrupt
266 // for HSync-Counter using Update
267 // for DMA Trigger START using CH3
268 //---------------------------------------------
269
270 TIM_ITConfig(TIM2,TIM_IT_CC3,ENABLE);
271
272 // NVIC config
273 NVIC_InitStructure.NVIC_IRQChannel = TIM2_IRQn;
274 NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0;
275 NVIC_InitStructure.NVIC_IRQChannelSubPriority = 0;
276 NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
277 NVIC_Init(&NVIC_InitStructure);
278}

281//--------------------------------------------------------------
282// internal Function
283// init DMA
284//--------------------------------------------------------------
285void P_VGA_InitDMA(void)
286{
287 DMA_InitTypeDef DMA_InitStructure;
288
289 //---------------------------------------------
290 // DMA of Timer1 Update
291 // (look at page 217 of the Ref Manual)
292 // DMA=2, Channel=6, Stream=5
293 //---------------------------------------------
294
295 // Clock Enable (DMA)
296 RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_DMA2, ENABLE);
297
298 // DMA init (DMA2, Channel6, Stream5)
299 DMA_Cmd(DMA2_Stream5, DISABLE);
300 DMA_DeInit(DMA2_Stream5);
301 DMA_InitStructure.DMA_Channel = DMA_Channel_6;
302 DMA_InitStructure.DMA_PeripheralBaseAddr = (uint32_t)VGA_GPIOE_ODR_ADDRESS;
303 DMA_InitStructure.DMA_Memory0BaseAddr = (uint32_t)&VGA_RAM1;
304 DMA_InitStructure.DMA_DIR = DMA_DIR_MemoryToPeripheral;
305 DMA_InitStructure.DMA_BufferSize = VGA_DISPLAY_X+1;
306 DMA_InitStructure.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
307 DMA_InitStructure.DMA_MemoryInc = DMA_MemoryInc_Enable;
308 DMA_InitStructure.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Byte;
309 DMA_InitStructure.DMA_MemoryDataSize = DMA_MemoryDataSize_Byte;
310 DMA_InitStructure.DMA_Mode = DMA_Mode_Normal;
311 DMA_InitStructure.DMA_Priority = DMA_Priority_VeryHigh;
312 DMA_InitStructure.DMA_FIFOMode = DMA_FIFOMode_Disable;
313 DMA_InitStructure.DMA_FIFOThreshold = DMA_FIFOThreshold_HalfFull;
314 DMA_InitStructure.DMA_MemoryBurst = DMA_MemoryBurst_Single;
315 DMA_InitStructure.DMA_PeripheralBurst = DMA_PeripheralBurst_Single;
316 DMA_Init(DMA2_Stream5, &DMA_InitStructure);
317
318 // DMA-Timer1 enable
319 TIM_DMACmd(TIM1,TIM_DMA_Update,ENABLE);
320}

324//--------------------------------------------------------------
325// Interrupt of Timer2
326//
327// CC3-Interrupt -> starts from DMA
328// Watch it.. higher troughput when interrupt flag is left alone
329//--------------------------------------------------------------
330void TIM2_IRQHandler(void)
331{
332
333 // Interrupt of Timer2 CH3 occurred (for Trigger start)
334 TIM_ClearITPendingBit(TIM2, TIM_IT_CC3);
335// TIM2->SR |= ~TIM_SR_CC3IF; //Clear pending bit interrupt
336// NVIC->ISPR[0] = 0x10000000;
337// NVIC->ICPR[0] = 0x10000000;
338// TIM2->SR = (uint16_t)~((uint16_t)0x0008);
339
340
341 VGA.hsync_cnt++;
342 if(VGA.hsync_cnt>=VGA_VSYNC_PERIODE) {
343 // -----------
344 VGA.hsync_cnt=0;
345 // Adresspointer first dot
346 VGA.start_adr=(uint32_t)(&VGA_RAM1[0]);
347 }
348
349 // HSync-Pixel
350 if(VGA.hsync_cnt<VGA_VSYNC_IMP) {
351 // HSync low
352 GPIOB->BSRRH = GPIO_Pin_12;
353 }
354 else {
355 // HSync High
356 GPIOB->BSRRL = GPIO_Pin_12;
357 }
358
359 // Test for DMA start
360 if((VGA.hsync_cnt>=VGA_VSYNC_BILD_START) && (VGA.hsync_cnt<=VGA_VSYNC_BILD_STOP)) {
361 // DMA2 init
362 DMA2_Stream5->CR=VGA.dma2_cr_reg;
363 // set address
364 DMA2_Stream5->M0AR=VGA.start_adr;
365 // Timer1 start
366 TIM1->CR1|=TIM_CR1_CEN;
367 // DMA2 enable
368 DMA2_Stream5->CR|=DMA_SxCR_EN;
369
370 // Test Adrespointer for high
371 if((VGA.hsync_cnt & 0x01)!=0) {
372 // inc after Hsync
373 VGA.start_adr+=(VGA_DISPLAY_X+1);
374 }
375 }
376
377}
